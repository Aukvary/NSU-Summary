##### add rn, rm
- rm += rn
- с: произошло ли беззнаковое переполнение(255 + 1 = 0)
- v: произошло ли знаковое переполнение(127 + 1 = -128)
- z: если результат равен 0
- n: если старший бит равен 1
##### addc rn, rm
- rm += rn + c
- с: произошло ли беззнаковое переполнение(255 + 1 = 0)
- v: произошло ли знаковое переполнение(127 + 1 = -128)
- z: если результат равен 0
- n: если старший бит равен 1
##### sub rn, rm
- rm = rn - rm
- c: если не было переполнения, то бит будет равен 1(150 - 100), а если было, то бит равен 0
- v: если знаковая операция дала переполнение(127 - -1 = -128)
- z: если результат равен 0
- n: если старший бит равен 1
##### cmp rn, rm
- вычисляет rn - rm, но не меняет значение регистров
- c: если не было переполнения, то бит будет равен 1(150 - 100), а если было, то бит равен 0
- v: если знаковая операция дала переполнение(127 - -1 = -128)
- z: если результат равен 0
- n: если старший бит равен 1
##### neg rn
- rn = -rn
- c: если не было переполнения, то бит будет равен 1(150 - 100), а если было, то бит равен 0
- v: если знаковая операция дала переполнение(127 - -1 = -128)
	- особый случай: -128, результат останется тем же и флаг установится в 1
- z: если результат равен 0
- n: если старший бит равен 1
##### inc rn
- rn += 1
- c: не меняет бит 
- v: произошло ли знаковое переполнение(127 + 1 = -128)
- z: если результат равен 0
- n: если старший бит равен 1
##### dec rn
- rn -= 1
- c: не меняет бит 
- v: если знаковая операция дала переполнение(127 - -1 = -128)
- z: если результат равен 0
- n: если старший бит равен 1
##### clr rn
- rn = 0
- c: 1
- v: 0
- z: 1
- n: 0
##### tst rn
- rn = rn
- c: 0
- v: 0
- z: зависит от rn
- n: зависит от значения
