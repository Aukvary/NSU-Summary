- создается объект сокета функцией socket с доменом af_inet и типом sock_stream
- сокет привязывается к ip-адресу и порту через bind
- устанавливается режим прослушивания и размер очереди ожидания функцией listen
- в цикле вызывается accept для извлечения новых соединений и получения дескрипторов для обмена данными
```C
#include <sys/socket.h>
#include <netinet/in.h>

void start_server(void) {
    int sd = socket(AF_INET, SOCK_STREAM, 0),
    struct sockaddr_in addr = { .sin_family = AF_INET, .sin_port = htons(8080) },
    bind(sd, (struct sockaddr*)&addr, sizeof(addr)),
    listen(sd, 10),
    while (1) {
        int client_sd = accept(sd, NULL, NULL),
        // обработка клиента в потоке или процессе
    }
}
```
