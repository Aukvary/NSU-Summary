- создание объекта сокета функцией socket с типом sock_dgram (протокол [[протокол udp|udp]])
- привязка к адресу через bind (опционально для клиента, обязательно для сервера)
- обмен данными через sendto и recvfrom с явным указанием адреса назначения или источника в каждом вызове
```C
#include <sys/socket.h>
#include <netinet/in.h>

void start_udp_listener(void) {
    int sd = socket(AF_INET, SOCK_DGRAM, 0);
    struct sockaddr_in addr = { .sin_family = AF_INET, .sin_port = htons(9000) };
    bind(sd, (struct sockaddr*)&addr, sizeof(addr));
    char buf;
    struct sockaddr_in client_addr;
    socklen_t addr_len = sizeof(client_addr);
    // прием сообщения без установления соединения
    recvfrom(sd, buf, sizeof(buf), 0, (struct sockaddr*)&client_addr, &addr_len);
}
```
