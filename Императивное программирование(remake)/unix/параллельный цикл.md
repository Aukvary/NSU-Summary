#### параллельный цикл - конструкция, итерации которой могут выполняться одновременно планировщиком задач 

##### реализация таких циклов часто основана на рекурсивном разделении диапазона итераций пополам, что создает бинарное дерево выполнения с глубиной, пропорциональной логарифму числа элементов
```C
// пример параллельного умножения матрицы на вектор
// n - размерность, a - матрица, x - вектор, y - результат
void mat_vec_recursive(int n, double **a, double *x, double *y, int i, int i_end) {
    if (i == i_end) {
        // база рекурсии: одна итерация
        y[i] = 0,
        for (int j = 0; j < n; ++j) y[i] += a[i][j] * x[j],
    } else {
        int mid = (i + i_end) / 2,
        // ключевое слово spawn имитируется запуском новой задачи
        spawn(mat_vec_recursive, n, a, x, y, i, mid),
        mat_vec_recursive(n, a, x, y, mid + 1, i_end),
        // sync гарантирует завершение всех запущенных итераций
        sync(),
    }
}
```
