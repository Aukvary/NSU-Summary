- определяется функция-обработчик с сигнатурой void(int)
- используется системный вызов signal для связывания номера сигнала с обработчиком
- процесс переходит в режим ожидания или продолжает работу до момента получения сигнала ядром
```C
#include <stdio.h>
#include <signal.h>
#include <unistd.h>

// обработчик сигналов
void handle_signal(int sig) {
    if (sig == SIGUSR1) printf("received sigusr1\n");
}

int main(void) {
    // регистрация обработчика
    signal(SIGUSR1, handle_signal);
    while (1) {
        pause(); // ожидание любого сигнала
    }
    return 0;
}
```
