#### иммутабельная очередь - структура данных, реализованная с помощью двух [[иммутабельный стек|иммутабельных стеков]], что обеспечивает [[персистентность|персистентность]]
##### один стек используется для вставки, другой для удаления, при опустошении стека удаления происходит переливание из стека вставки, что дает амортизированную сложность $o(1)$
#### сигнатура
```C
struct node {
    int value;
    struct node *next;
};

struct p_stack {
    struct node *top;
};

struct p_queue {
    struct p_stack* inbox;
    struct p_stack* outbox;
};
```
#### вставка
```C
struct p_queue push(struct p_queue q, int val) {
    struct p_queue new_q = {push(q.inbox, val), q.outbox};
    return new_q;
}
```

#### удаление
```C
struct p_queue pop(struct p_queue q) {
    if (q.outbox) {
	    stack_pop(q.outbox);
	    
	    return (p_queue){q.inbox, q.outbox};
    }
    
    while (q.inbox) {
	    stack_push(q.outbox, stack_pop(q.inbox));
    }
    
    return (p_queue) {q.inbox, q.outbox};
}
```
