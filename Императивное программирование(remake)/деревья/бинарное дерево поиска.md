#### бинарное дерево поиска - [[бинарное дерево]], где левый потомок меньше текущего узла, а правый больше 

##### определение характеристик
- **высота:** вычисляется рекурсивно: $1 + \max(\text{left}, \text{ right})$ 
- **общее количество узлов:** равно сумме узлов в левом и правом поддеревьях плюс 1
- **внешние узлы(листья):** если узел не имеет детей, он считается внешним(возвращается 1), иначе это сумма внешних узлов его потомков
- **внутренние узлы:** узлы, имеющие хотя бы одного потомка

#### сигнатура
```C
typedef struct Node {
	int val;
	
	struct Node* left;
	struct Node* right;
} Node;
```
#### вставка
```C
void insert(Node* root, Node* new) {
	if (root->val == new->val) {
		return;
	} else if (root->val > new->val) {
		if (root->left) {
			insert(root->left, new);
		} else root->left = new;
	} else {
		if (root->right) {
			insert(root->right, new);
		} else root->right = new;
	}
}
```
#### поиск
```C
Node* search(Node* node, int val) {
	if (!node) return NULL;
	if (node->val == val) return node;
	if (node->val > val) return search(node->left, val);
	return search(node->right, val);
}
```
#### удаление
```C
Node* remove(Node* node, int val) {
	Node* removed = search(node, val);
	if (!removed) return NULL;
	
	if (!removed->left && !removed->right) {
		return removed;
	}
	if (removed->left && !removed->right) {
		removed->val = removed->left->val;
		removed->count = removed->left->count;
		return removed->left;
	}
	if (!removed->left && removed->right) {
		removed->val = removed->right->val;
		removed->count = removed->right->count;
		return removed->right;
	}
	
	Node* maxMin = node->left;
	
	while (maxMin->right) {
		maxMin = maxMin->right;
	}
	
	removed->val = maxMin->right->val;
	removed->count = maxMin->right->count;
	return maxMin;
}
```
