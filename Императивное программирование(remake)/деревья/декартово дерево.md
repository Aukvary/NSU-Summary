- структура данных, которая объединяет свойства бинарной кучи и [[бинарное дерево поиска|бинарного дерева поиска]] 
- каждый узел содержит 2 значения $(k, p)$, где 
	- $k$ - основной ключ
	- $p$ - приоритет узла
- свойства
	- относительно $k$: для любого узла ключи всех элементов в левом поддереве меньше $k$, а в правом больше 
	- относительно $p$: приоритет любого узла должен быть выше, чем приоритет его дочерних узлов
- чтобы гарантировать сбалансированность, приоритеты $p$ выбираются как случайные числа, что даёт равномерное распределение без поворотов
##### сигнатура
```C
typedef struct node {
	struct node* left;
	struct node* right;
	
	int key;
	int priority;
} node;

void split(node* t, int key, node** l, node** r) {
	if (!t) {
		*l = NULL;
		*r = NULL;
	} else if (key < t->key) {
		split(t->left, key, l, &(t->left));
		*r = t;
	} else {
		split(t->right, key, &(t->right), r);
		*l = t;
	}
}

void merge(node* l, node* r) {
	if (!l || !r) {
		return l ? l : r;
	}
	
	if (l->priority > r->priority) {
		l->right = merge(l->right, r);
		return l;
	}
	
	r->left = merge(l, r->left);
	return r;
}
```
##### вставка
```C
void insert(node* head, node* val) {
	node* l;
	node* r;
	split(head, val->key, &l, &r);
	return merge(merge(l, val), r);
}
```

##### удаление
```C
node* remove(node* head, int key) {
	node* l;
	node* r;
	split(head, key - 1, &l, &r);
	
	node* n;
	node* nr;
	
	split(head, key, &n, &r);
	
	merge(l, r);
	return n;
}
```
