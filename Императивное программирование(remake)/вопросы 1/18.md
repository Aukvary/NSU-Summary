#### Что такое рекурсия, и каковы её преимущества и недостатки? Приведите примеры рекурсивных функций, таких как вычисление факториала или перевод числа в двоичную систему
- [[рекурсия]]
```C
int fac(int n) {
	if (n == 0 || n == 1) return 1; 
	
	return n * fac(n - 1);
}

void bin(int n, int* res, int* idx) {
	if (n == 0) return;
	res[(*idx)++] = n / 2;
	n /= 2;
	bin(n, res, idx)
}
```
