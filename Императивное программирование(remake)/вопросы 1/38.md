#### Дайте определение динамического программирования, его основные принципы и отличия от жадных алгоритмов. На примере задачи взвешенного интервального планирования опишите переход от рекурсивного решения с экспоненциальной сложностью к эффективному алгоритму с использованием мемоизации или итеративного заполнения таблицы
- [[динамическое программирование]]
```C
#include <stdlib.h>
#include <string.h>

typedef struct {
	int start, final, weight;
} Req;

int cmp(const void* a, const void* b) {
	return ((Req*)a)->final - ((Req*)b)->final;
}

int func(Req* reqs, int count) {
	qsort(reqs, count, sizeof(Req), cmp);
	
	int* p = calloc(sizeof(int), count);
	for (int i = 0; i < count; i++) { p[i] = -1; }
	
	for (int i = 1; i < count; i++) {
		for (int j = count - i; i >= 0; j--) {
			if (reqs[i].start < reqs[j].final)
				continue;
		
			p[i] = j;
			break;
		}
	}
	
	int* opt = calloc(sizeof(int), count);
	opt[0] = reqs[0].weight;
	
	for (int i = 1; i < count; i++) {
		int a = reqs[i].weight + (p[i] == -1 ? 0 : opt[p[i]]);
		int b = opt[i - 1];
	
		opt[i] = a < b ? b : a;
	}
	
	int res = opt[count - 1];
	
	free(p);
	free(opt);
	
	return res;

}
```
