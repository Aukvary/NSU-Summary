- каждый участок разбивается на $n$ парабол, где $n$ обязательно чётное
```C
typedef float(*func)(float);

float simpson_area(float a, float b, func f, int n) {
	if (n % 2) {
		n++;
	}
	
	float w = (b - a) / n;
	float sum = f(a) + f(b);
	
	for (int i = 1; i < n; i++) {
		float x = a + w * i;
		
		if (i % 2) {
			sum += 4.0f * f(x);
		} else {
			sum += 2.0f * f(x);
		}
	}
	
	return sum * w / 3;
}

float df(float x, func f) {
	float h = 1e-4f;
	float dif = (f(x + h) - f(x - h)) / (2.0f * h);

	return sqrtf(1.0f + dif * dif);
}

  

float simpson_len(float a, float b, func f, int n) {
	if (n % 2 != 0) {
		n++;
	}
	
	float w = (b - a) / n;
	float sum = df(a, f) + df(b, f);
	
	for (int i = 1; i < n; i++) {
		float x = a + w * i;
		
		if (i % 2) {
			sum += 4.0f * df(x, f);
		} else {
			sum += 2.0f * df(x, f);
		}
	}
	
	return sum * w / 3;
}
```
