- делим функцию на $n$ прямых, если соединить их концы, то получатся прямоугольные трапеции
- чтобы посчитать площадь достаточно сложить все площади трапеций
- чтобы посчитать длину достаточно сложить длины отрезков, соединяющих прямые 
```C
typedef float(*func)(float);

float trap_area(float a, float b, func f, int n) {
	float w = (b - a) / n;
	float sum = f(a) + f(b);
	
	for (int i = 1; i < n; i++) {
		float x = a + w * i;
		sum += 2.0f * f(x);
	}
	
	return sum * w / 2;
}

float rect_len(float a, float b, func f, int n) {
	float w = (b - a) / n;
	float sum = 0;
	
	for (int i = 0; i < n; i++) {
		float x1 = a + w * i;
		float x2 = x1 + w;
		
		float dy = f(x2) - f(x1);
		sum += sqrt(w * w + dy * dy);
	}
	
	return sum;
}
```
