- делим функцию на $n$ прямоугольников и рассматриваем функцию, как идущие подряд прямоугольники
- для площади графика суммируем площади этих прямоугольников
- для длинны графика мы суммируем расстояние между вершинами прямоугольников
```C
typedef float(*func)(float);

float rect_area(float a, float b, func f, int n) {
	float w = (b - a) / n;
	float sum = 0;
	
	for (int i = 0; i < n; i++) {
		float x = a + w * i;
		sum += f(x);
	}
	
	return sum * w;
}

float rect_len(float a, float b, func f, int n) {
	float w = (b - a) / n;
	float sum = 0;
	
	for (int i = 0; i < n; i++) {
		float x1 = a + w * i;
		float x2 = x1 + w;
		float dy = f(x2) - f(x1);
		sum += sqrt(w * w + dy * dy);
	}
	
	return sum;
}
```
