void counting_sort(int* nums, int n) {	
	int max = nums[0]; 
	for (int i = 1; i < n; i++) { 
		if (nums[i] > max) { 
			max = nums[i]; 
		} 
	}
	
	int* arr = calloc(max + 1, sizeof(int));
	int* temp = malloc(n * sizeof(int));
	
	for (int i = 0; i < n; i++) {
		arr[nums[i]]++;
	}
	
	for (int i = 1; i < n; i++) {
		arr[i] += arr[i - 1];
	}
	
	for (int i = 0; i < n; i++) {
		temp[arr[nums[i]]--] = nums[i];
	}
	
	for (int i = 0; i < n; i++) {
		nums[i] = temp[i];
	}
	
	free(arr);
	free(temp);
}
