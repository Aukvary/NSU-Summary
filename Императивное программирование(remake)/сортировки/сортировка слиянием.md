- делим массив пополам пока не дойдём до размера минимального размера(0-2)
- объединяем части массива 
```C
int* temp; // инициализирован размером массива

void merge_sort(int* nums, int len) {
	if (len < 2) return;

	int l = len / 2;
	int r = len - l;
	
	int* left = nums;
	int* right = nums + l;
	
	merge_sort(left, l);
	merge_sort(right, r);
	
	int tidx = 0;
	int lidx = 0;
	int ridx = 0;
	while (lidx < l && ridx < r) {
		if (left[lidx] < right[ridx]) {
			temp[tidx++] = left[lidx++];
		} else {
			temp[tidx++] = right[ridx++];
		}
	}
	
	while (lidx < l) {
		temp[tidx++] = left[lidx++];
	}
	
	while (ridx < r) {
		temp[tidx++] = right[ridx++];
	}
	
	for (int i = 0; i < len; i++) { 
		nums[i] = temp[i]; 
	}
}
```
