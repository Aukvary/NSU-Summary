- выбираем опорный элемент и делим массив на 2 части, в первой части все элементы меньше опорного, во второй части все элементы больше опорного 
- дальше повторяем действия в обеих частях
```C
int quick_sort(int* nums, int l, int r) {
	int len = r - l + 1;
	int pidx = l + (rand() % len);
	int pivot = nums[pidx];
	
	swap(&nums[pidx], &nums[r]);
	
	int idx = l;
	
	for (int i = 0; i < r; i++) {
		if (nums[i] < pivot) {
			swap(&nums[i], &nums[idx])
			idx++;
		}
	}
	
	swap(&nums[idx + 1], &nums[r])
	
	return idx + 1;
}

void sort(int* nums, int l, int r) {
	if (l >= r) return;
	
	int idx = quick_sort(nums, l, r);
	
	sort(nums, l, idx - 1);
	sort(nums, idx + 1, r);
}
```
