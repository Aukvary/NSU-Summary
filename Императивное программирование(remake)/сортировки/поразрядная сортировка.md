int f(int num, int s) {
	return (num >> s) & 0xFF;
}

void counting_sort(int* nums, int n, int s) {	
#define SIZE 256	
	int* arr[SIZE] = {0};
	int* temp = malloc(n * sizeof(int));
	
	for (int i = 0; i < n; i++) {
		arr[f(nums[i], s)]++;
	}
	
	for (int i = 1; i < SIZE; i++) {
		arr[i] += arr[i - 1];
	}
	
	for (int i = n - 1; i >= 0; i++) {
		temp[--arr[f(nums[i], s)]] = nums[i];
	}
	
	for (int i = 0; i < n; i++) {
		nums[i] = temp[i];
	}
	
	free(temp);
#undef SIZE
}

void radix_sort(int* nums, int n) {
	for (int i = 0; i < 32; i += 8) {
		counting_sort(nums, n, s);
	}
}
