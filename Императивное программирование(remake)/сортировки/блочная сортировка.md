#include <stdlib.h>

typedef struct Node {
	float value;
	struct Node* next;
} Node;

  

Node* insert(Node* head) {
	if (!head || !head->next)
		return head;

	Node* sorted = NULL;
	Node* cur = head;
	
	while (cur != NULL)
		Node* next = cur->next;
		
		if (!sorted || sorted->value >= cur->value) {
			cur->next = sorted;
			sorted = cur;
		} else {
			Node* temp = sorted;
			while (temp->next != NULL && temp->next->value < cur->value) {
				temp = temp->next;
			}
		
			cur->next = temp->next;
			temp->next = cur;
		}
		cur = next;
	}
	
	return sorted;
}

  

void bucket_sort(float* nums, int n) {
	Node** buckets = calloc(n, sizeof(Node*));

	for (int i = 0; i < n; i++) {
		int idx = (int)(n * nums[i]);
		Node* new = malloc(sizeof(Node));
		new->value = nums[i];
		
		new->next = buckets[idx];
		buckets[idx] = new;
	}

	int arri = 0;
	for (int i = 0; i < n; i++) {
		Node* cur = insert(buckets[i]);
		
		while (cur != NULL) {
			nums[arri++] = cur->value;
			void* tmp = cur;
			cur = cur->next;
			free(tmp);
		}
	}

	free(buckets);

}
