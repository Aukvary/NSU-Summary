int heapify(int* nums, int len, int i) {
	int idx = i;
	int l = i * 2 + 1;
	int r = i * 2 + 2;
	
	if (nums[idx] < nums[l]) {
		idx = l;
	} 
	
	if (nums[idx] < nums[r]) {
		idx = r;
	}
	
	if (idx != i) {
		swap(&nums[i], &nums[idx]);
		heapify(nums, len, idx);
	}
}

void heap_sort(int* nums, int len) {
	for (int i = len / 2 - 1; i >= 0; i--) {
		heapify(nums, leb, i);
	}

	for (int i = 0; i < len; i++) {
		swap(&nums[0], &nums[len - i - 1]);
		heapify(nums, len - i);
	}
}
