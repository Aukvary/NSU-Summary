#### алгоритм
- итеративно ищет [[Императивное программирование(remake)/графы/увеличивающий путь]] в [[остаточная сеть|остаточной сети]]
- находит минимальную [[остаточная пропускная способность|остаточную пропускную способность]] вдоль пути
- увеличивает общий поток на найденную величину и обновляет остаточную сеть
- повторяет процесс до тех пор, пока путь из истока в сток не исчезнет
```C
// структура ребра для представления сети
struct edge {
    int to;
    int capacity;
    int flow;
    int rev; // индекс обратного ребра
};

// упрощенная сигнатура функции поиска максимального потока
// сложность o(e * |f_max|), где f_max - величина макс. потока
// применяется в транспортных задачах и задачах о назначениях
int ford_fulkerson(int s, int t, int n) {
    int max_flow = 0;
    while (1) {
        // поиск увеличивающего пути p (например, через dfs)
        // если путь p не найден, break
        
        // вычисление cf(p) = min(cf(u,v) на пути)
        int path_flow = find_min_residual_capacity(p);
        
        // обновление потоков на ребрах пути
        for (int i = 0; i < path_length; ++i) {
            edge *e = get_edge(p, i);
            e->flow += path_flow;
            get_reverse_edge(e)->flow -= path_flow;
        }
        max_flow += path_flow;
    }
    return max_flow;
}
```
