#### алгоритм
- итеративно ищет [[Императивное программирование(remake)/графы/увеличивающий путь|увеличивающий путь]] в [[остаточная сеть|остаточной сети]] 
- находит минимальную [[остаточная пропускная способность|остаточную пропускную способность]] вдоль пути
- увеличивает общий поток на найденную величину и обновляет остаточную сеть
- повторяет процесс до тех пор, пока путь из истока в сток не исчезнет
```C
int** graph;
int n;

int* path;

int* queue;
bool* visited;

bool bfs(int s, int t) {
	memset(visited, 0, sizeof(bool) * n);
	int head = 0;
	int tail = 0;
	
	queue[tail++] = s;
	
	while (head < tail) {
		int u = queue[head++];
		
		for (int i = 0; i < n; i++) {
			if (graph[u][i] == 0 || visited[i]) continue;
			path[i] = u;
			
			if (i == t) {
				return true;
			}
			
			visited[i] = true;
			queue[tail++] = i;
		}
	}
	
	return false;
}

int max_flow(int s, int t) {
	int mf = 0;
	while(bfs(s, t)) {
		int min = graph[path[t]][t];
		for (int i = path[t]; i != s; i = path[i]) {
			min = MIN(min, graph[path[i]][i]);
		}
		
		graph[path[t]][t] -= min;
		
		for (int i = path[t]; i != s; i = path[i]) {
			int u = path[i];
			graph[u][i] -= min;
			graph[i][u] += min;
		}
		
		mf += min;
	}
	
	return mf;
}
```
