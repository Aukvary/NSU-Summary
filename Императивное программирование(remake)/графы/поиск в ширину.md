- позволяет находить кратчайший путь в невзвешенном графе 
```C
void bfs(int** graph, int start, int n) {
	int* queue = malloc(sizeof(int) * n);
	bool visited = calloc(sizeof(bool),  n);
	int head = 0, tail = 0;
	
	queue[tail++] = start;
	visited[start] = true;
	while (head < tail) {
		int u = queue[head++];
		
		for (int i = 0; i < n; i++) {
			if (graph[u][i] && !visited[i]) {
				queue[tail++] = i;
				visited[i] = true;
			}
		}
		
		printf("%d:", u);
	}
	
	free(queue);
	free(visited);
}
```
