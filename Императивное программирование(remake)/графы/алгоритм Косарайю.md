int n;
int** graph;
int** tgraph;
bool* visited;

int* tin;
int* tin_p = 0;

int* component;
int count;

void dfs1(int u) {
	visited[u] = true;
	
	for (int i = 0; i < n; i++) {
		if (graph[u][i] && !visited[i]) {
			dfs1(i);
		}
	}
	
	tin[tin_p++] = u;
}

void dfs2(int u, int c) {
	component[u] = c;
	
	for (int i = 0; i < n; i++) {
		if (tgraph[u][i] && component[i] == -1) {
			dfs2(i, c);
		}
	}
}

void find() {
	for (int i = 0; i < n; i++) { 
		component[i] = -1; 
		visited[i] = false; 
	}


	for (int i = 0; i < n; i++) {
		if (!visited[i]) dfs1(i);
	}
	
	for (int i = n - 1; i >= 0; i--) {
		int u = tin[i];
		if (!component[u]) {
			dfs2(u, count);
			count++;
		}
	}
}
