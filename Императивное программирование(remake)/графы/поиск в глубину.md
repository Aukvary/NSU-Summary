void dfs(int** graph, int start, int n) {
	int* stack = malloc(sizeof(int) * n);
	bool* visited = calloc(sizeof(bool), n);
	int ptr = 0;
	
	stack[ptr++] = start;
	visited[start] = true;
	
	while(ptr) {
		int u = stack[ptr--];
		
		if (visited[u]) { continue; } 
		
		visited[u] = true; 
		printf("%d: \n", u);
		
		
		for (int i = 0; i < n; i++) {
			if (graph[u][i] && !visited[i]) {
				stack[ptr++] = i;
			}
		}
	}
}
