##### 邪谢谐芯褉懈褌屑 锌芯褋谢械写芯胁邪褌械谢褜薪芯 胁褘蟹褘胁邪械褌 锌褉芯褑械写褍褉褍 褉邪蟹谐褉褍蟹泻懈 (discharge) 写谢褟 胁械褉褕懈薪 褋锌懈褋泻邪, 锌芯泻邪 薪械 斜褍写械褌 写芯褋褌懈谐薪褍褌 泻芯薪械褑 褋锌懈褋泻邪, 褔褌芯 芯蟹薪邪褔邪械褌 芯褌褋褍褌褋褌胁懈械 锌械褉械锌芯谢薪械薪薪褘褏 褍蟹谢芯胁

// 褋谢芯卸薪芯褋褌褜 胁褋械泄 褎邪蟹褘 o(v^3) 蟹邪 褋褔械褌 褌芯锌芯谢芯谐懈褔械褋泻芯谐芯 褍锌芯褉褟写芯褔懈胁邪薪懈褟
void discharge(int u) {
    while (excess[u] > 0) {
        int v = current_neighbor[u],
        if (v == -1) {
            relabel(u),
            current_neighbor[u] = head[u],
        } else if (res_cap(u, v) > 0 && h[u] == h[v] + 1) {
            push(u, v),
        } else {
            current_neighbor[u] = next[v],
        }
    }
}
