int* top_sort(int** graph, int n) {
	int** queue = malloc(sizeof(int) n);
	int head = 0, tail = 0;
	
	int* counts = calloc(sizeof(int), n);
	
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			count[i] += graph[j][i];
		}
	}
	
	for (int i = 0; i < n; i++) {
		if (counts[i] == 0) {
			queue[tail++] = i;
		}
	}
	
	int* res = malloc(sizeof(int) * n)
	int ptr = 0;
	
	while (head < tail) {
		int u = queue[head++];
		res[ptr++] = u;
		
		for (int i = 0; i < n; i++) {
			if (graph[u][i]) {
				counts[i]--;
				if (!counts[i]) queue[tail++] = i;
			}
		}
	}
	
	return res;
}
