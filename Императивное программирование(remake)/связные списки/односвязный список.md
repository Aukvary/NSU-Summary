- [[связный список]], где каждый узел ссылается только на следующий узел
#### сигнатура
```C
typedef struct List {
	struct List* next;
	int val;
} List;
```
#### вставка
```C
void insert(List* list, List* value) {
	while(list->next) {
		list = list->next;
	}
	
	list->next = value;
}
```
#### поиск
```C
List* search(List* list, int val) {
	while(list) {
		if (list->val == val) 
			return list;
			
		list = list->next;
	}
	
	return NULL;
}
```
#### удаление
```C
List* remove(List* list, int val) {
	List* prev = NULL;
	
	while(list) {
		if (list->val == val) {
			if (prev) {
				prev->next = list->next;
			}
			
			return list;
		}
		prev = list;
		list = list->next;
	}
	
	return NULL;
}
```
