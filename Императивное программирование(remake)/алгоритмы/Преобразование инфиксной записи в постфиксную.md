- $(A + B) \rightarrow (AB+)$ 
```C
char* stack;
int ptr = -1;

char* transform(const char* line) {
	int len = strlen(line);
	char* res = malloc(len + 1);
	int resi = 0;
	stack[++ptr] = '(';
	for (int i = 0; i <= len; i++) {
		char symb = (i == len ? ')' : line[i]);
		
		if (symb == '(') {
			stack[++ptr] = '(';
		} else if (symb == ')') {
			while (ptr >= 0 && stack[ptr] != '(') {
				res[resi++] = stack[ptr--];
			}
			if (ptr >= 0) ptr--;
		} else if (symb == '-' || symb == '+') {
			while (ptr >= 0 && (stack[ptr] == '-' || stack[ptr] == '+' ||
								stack[ptr] == '*' || stack[ptr] == '/')) {
				res[resi++] = stack[ptr--];
			}
			stack[++ptr] = symb;
		} else if (symb == '*' || symb == '/') {
			while (ptr>= 0 && (stack[ptr] == '*' || stack[ptr] == '/')) {
				res[resi++] = stack[ptr--];
			}
			stack[++ptr] = symb;
		} else {
			res[resi++] = symb;
		}
	}
	
	res[resi] = 0;
	return res;
}
```
