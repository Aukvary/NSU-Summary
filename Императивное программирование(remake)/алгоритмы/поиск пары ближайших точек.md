typedef struct {
	float x;
	float y;
} Vec2;

int cmpx(const void* a, const void* b) {
	return ((Vec2*)a)->x - ((Vec2*)b)->x;
}

int cmpy(const void* a, const void* b) {
	return ((Vec2*)a)->y - ((Vec2*)b)->y;
}

float dist(Vec2 p1, Vec2 p2) {
	return (p1.x - p2.x) * (p1.x - p2.x) + (p1.y - p2.y) * (p1.y - p2.y);
}

float closes(Vec2* points_x, points_y, int n) {
	if (n <= 3) {
		float md = INFINITY;
		for (int i = 0; i < n; i++) {
			for (int j = i + 1; j < n; j++) {
				md = MIN(md, dist(points[i], points[j]));
			}
		}
		
		return md;
	}
	
	int mid = n / 2;
	
	int pivot = points_x[mid];
	
	int left = malloc(sizeof(Vec2) * mid);
	int right = malloc(sizeof(Vec2) * (mid - n));
	
	for (int i = 0, li = 0, ri = 0; i < n; i++) {
		if (points_y[i].x <= pivot && li <= mid) {
			left[li++] = points_y[i];
		} else {
			right[ri++] = points_y[i];
		}
	}
	
	float dl = closes(points_x, left, mid);
	float dr = closes(points_x + mid, right, n - mid);
	
	float d = MIN(dl, dr);
	
	Vec2* strip = malloc(sizeof(Vec2) * n);
	int size = 0; 
	
	for (int i = 0; i < n; i++) { 
		if (fabsf(points_y[i].x - pivot.x) < d) { 
			strip[size++] = points_y[i]; 
		} 
	} 
	
	for (int i = 0; i < size; i++) { 
		for (int j = i + 1; j < size && (strip[j].y - strip[i].y) < d; j++) { 
			d = MIN(d, dist(strip[i], strip[j])); 
		} 
	} 
	
	free(left); 
	free(right); 
	free(strip); 
	return d;
}

float func(Vec2* points, int n) {
	Vec2* points_x = malloc(sizeof(Vec2) * n); 
	Vec2* points_y = malloc(sizeof(Vec2) * n); 
	for (int i = 0; i < n; i++) { 
		points_x[i] = points[i]; 
		points_y[i] = points[i]; 
	} 
	
	qsort(points_x, n, sizeof(Vec2), cmpx); 
	qsort(points_y, n, sizeof(Vec2), cmpy); 
	float result = closest_util(points_x, points_y, n); 
	free(points_x); 
	free(points_y); 
	return result;
}
