- инверсия - пара чисел в последовательности $a_1, a_2, ..., a_n$, где $i < j$ и $a_i > a_j$ 
- идея заключается в использовании модифицированного алгоритма [[сортировка слиянием|сортировки слиянием]] 
- когда мы соединяем 2 подмассива, если мы берём элемент из правой части это значит, что он меньше $n$ оставшихся элементов в правой части -> для него существует $n$ инверсий 
```C
int* temp; // инициализирован размером массива

int merge_sort(int* nums, int len) {
	if (len < 2) return 0;
	
	int l = len / 2;
	int r = len - l;
	int* left = nums;
	int* right = nums + l;
	
	int count = merge_sort(left, l) + merge_sort(right, r);
	
	int tidx = 0;
	int lidx = 0;
	int ridx = 0;
	
	while (lidx < l && ridx < r) {
		if (left[lidx] < right[ridx]) {
			temp[tidx++] = left[lidx++];
		} else {
			temp[tidx++] = right[ridx++];
			count += l - lidx;
		}
	}
	
	while (lidx < l) {
		temp[tidx++] = left[lidx++];
	}
	
	while (ridx < r) {
		temp[tidx++] = right[ridx++];
	}
	
	for (int i = 0; i < len; i++) {
		nums[i] = temp[i];
	}
	
	return count;
}
```
