int* stack;
int ptr = -1;
int calc(const char* line) {
	for (int i = 0; line[i] != 0; i++) {
		if (line[i] == '+' || line[i] == '-' || line[i] == '*' || line[i] == '/') {
			int a = stack[ptr--];
			int b = stack[ptr--];
			
			switch (line[i]) {
				case '+': {
					stack[++ptr] = a + b;
					break;
				}
				case '-': {
					stack[++ptr] = b - a;
					break;
				}
				case '*': {
					stack[++ptr] = a * b;
					break;
				}
				case '/': {
					stack[++ptr] = b / a;
					break;
				}
			}
		} else stack[++ptr] = line[i] - '0';
	}
	
	return stack[0];
}
