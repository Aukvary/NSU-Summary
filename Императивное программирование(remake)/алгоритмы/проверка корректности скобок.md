char* stack;
int ptr = 0;

bool func(const char* line) {
	for (int i = 0; line[i] != '\0'; i++) {
		if (line[i] == '(' || line[i] == '[' || line[i] == '{') {
			stack[ptr++] = line[i];
			continue;
		} 
		
		if (ptr == 0) return false;
		char s = stack[--ptr];
		
		if (s == '(' && line[i] != ')') return false;
		if (s == '[' && line[i] != ']') return false;
		if (s == '{' && line[i] != '}') return false;
	}
	
	return ptr == 0;
}
