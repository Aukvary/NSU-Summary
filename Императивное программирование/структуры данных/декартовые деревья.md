## сигнатура
```C
typedef struct node {
	int key;
	int priority;
	
	struct node* left;
	struct node* right;
} node;

void split(node* n, int key, node** l, node** r) {
	if (n == NULL) {
		*l = NULL;
		*r = NULL;
		return;
	}
	
	if (n->key <= key) {
		*l = n;
		split(n->right, key, &((*l)->right), r);
	} else {
		*r = n;
		split(n->left, key, l, &((*r)->left));
	}
}

node* merge(node* l, node* r) {
	if (l == NULL) return r;
	if (r == NULL) return l;
	
	if (left->priority > right->priority) {
		l->right = merger(left->right, right);
		return l;
	} else {
		r->left = merge(l, r->left);
		return r;
	}
}
```
## вставка
```C
void push(node** head, node* n) {
	node* l;
	node* r;
	
	split(*head, n->key, &l, &r);
	
	*head = merger(merge(l, n), r);
}
```
## поиск
```C
node* search(Node* root, int key) {
    while (root != NULL) {
        if (key == root->key) {
            return root; 
        } else if (key < root->key) {
            root = root->left; 
        } else {
            root = root->right; 
        }
    }
    return NULL;  
}
```
## удаление
```C
void pop(head**, int key) {
	node* l;
	node* r;
	node* m;
	
	split(*head, key, &l, &r);
	split(l, key - 1, &l, &m);
	
	*head = merge(l, r);
}
```
