int rk_substr_search(char* str, char* target) {
	size_t target_hash = get_hash(target);
	size_t len = strlen(target);
	
	size_t cur_hash = 0;
	
	int count = 0;
	
	for (int i = 0; str[i] != '\0' && i < len; i++) {
		cur_hash = include_hash(cur_hash, str[i]);
	}
	
	int i;
	
	for (i = len; str[i] != '\0'; i++) {
		if (cur_hash == target_hash && strncmp(str + i, target, len)) {
			count += 1;
		}
		
		cur_hash = exclude_hash(cur_hash, str[i-len]);
		cur_hash = include_hash(cur_hash, str[i])
	}
	
	if (cur_hash == target_hash && strncmp(str + i, target, len)) {
		count += 1;
	}
	
	return count;
}
