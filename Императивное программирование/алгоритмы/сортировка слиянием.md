```C
void sort(int* nums, int len) {
    if (len < 2) return; // если массив пустой или 1 элмент, ничего не делаем, тк он уже отсортирован

	//делим массив попалам
    int len_l = len / 2; //длина левой части
    int len_r = len - len_l; //длина правой части

    int* nums_l = nums; // указатель на левую часть
    int* nums_r = nums + len_l; //указатель на правую часть

    sort(nums_l, len_l); //рекурсивно сортируем левую часть
    sort(nums_r, len_r); //рекурсивно сортируем правую часть

    int* temp = calloc(len, sizeof(int)); //врменный массив
    int i = 0, l_i = 0, r_i = 0;

	//заполняем массив в засимости оттого элемент какого массива(левого или правого) меньше
    while (l_i < len_l && r_i < len_r) {
        if (nums_l[l_i] <= nums_r[r_i]) {
            temp[i++] = nums_l[l_i++];
        } else {
            temp[i++] = nums_r[r_i++];
        }
    }
	//заполняем оставшееся
    while (l_i < len_l) {
        temp[i++] = nums_l[l_i++];
    }
    
    while (r_i < len_r) {
        temp[i++] = nums_r[r_i++];
    }
    //переносим значения из временнго массива в основной
    for (int i = 0; i < len; i++) {
        nums[i] = temp[i];
    }
    free(temp);
}
```

##### асимптотика
- скорость - O($N\log N$)
- память - O($N$)
