```C
void bin_seach(int* nums, int len, int target) {
	int l = 0; //левый указатель
	int r = len - 1; //правый указатель
	
	while (l <= r) { //пока левый указатель меньше правого
		int m = l + (r - l) / 2 //находим элемент между левым и правым 
		
		if (nums[m] == target) return m //если значение найдено возвращаем m
		// если значение меньше требуемого,  сдвигаем левый указатель
		else if (nums[m] < target) l = m + 1 
		// если значение больше требуемого, сдвигаем правый указатель
		else r = m - 1
	}
	//значение не найдено
	return -1;
}
```

##### асимптотика
- скорость - O($\log N$)
- память - O($1$)
