```C
int partition(int* nums, int low, int high) {
    int pivot = nums[high]; //опорная точка, относительно которой будет сортировка
    int i = low - 1; // начало индексации

    for (int j = low; j < high; j++) { //проходимся по всем значениям в диапозоне  [low; high)
        if (nums[j] <= pivot) { //если элементы меньше или равен опорной точке:
            i++; //сдвигаем указатель
            swap(&nums [i], &nums[j]); //меняем местами значения
        }
    }

    swap(&nums[i + 1], &nums[high]); 
    return i + 1;
}

void sort(int* nums, int low/*нижняя граница*/, int high/*верхняя граница*/) { 
    if (low < high) {
        int pi = partition(nums, low, high); //индексе элемента относильно, которого проводилась сортировка

        sort(nums, low, pi - 1); //рекурсивно сортируем леувую часть
        sort(nums, pi + 1, high); //рекурсивно сортируем правую часть
    }
}

void quick_sort(int* nums, int len) { //обёртка для удобства
	sort(nums, 0, len - 1)
}
```

##### асимптотика
- скорость - O($N\log N$) - O($N^2$ )
- память - O(1)
