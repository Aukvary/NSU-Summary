- Быстрая проверка очень больших чисел на простоту
- основан на [[малая теорема ферма|малой теореме ферма]]:
	- если $n$ простое, то $a^{n-1} \equiv 1 (\text{mod } n)$ 
	- представляем $n-1 = 2^s \cdot d$, где d нечётно
	- проверяем условие: либо $a^d \equiv 1 (\text{mod }n)$, либо существует такое $r < s$, что $a^{2^r \cdot} \equiv 1 (\text{mod } n)$ 
	- если условия не выполняются, то число точно составное, если выполняются, то оно вероятно простое
```C
typedef unsigned long long u64;

u64 mod_power(u64 a, u64 b, u64 m) {
	u64 res = 1;
	a %= m;
	
	while (b > 0) {
		if (b % 2 == 1) {
			res = (__int128)res * a % m;
		}
		
		a = (__int128)a * a % m;
		b /= 2;
	}
	
	return res;
}

bool miller_rabbin(u64 n, int k) {
	if (n < 4) return n != 1;
	if (n % 2 == 0) return false;
	
	u64 d = n - 1;
	int s = 0;
	
	while (d % 2 == 0) {
		d /= 2;
		s++;
	}
	
	for (int i = 0; i < k; i++) {
		u64 a = 2 + rand() % (n - 4);
		u64 x = mod_power(a, d, n);
		if (x == 1 || x == n - 1) continue;
		
		bool composite = true;
		for (int r = 1; r < ; r++) {
			x = (__int128)x*x % n;
			if (x == n - 1) {
				composite = false;
				break;
			}
		}
		if (composite) return false;
	}
	
	return true;
}
```
