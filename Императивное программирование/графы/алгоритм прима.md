- используется для построения минимального остовного дерева
- начинаем с произвольной вершины
- на каждом шаге выбираем ребро с минимальным весом, которое соединяет вершину, которая уже была, и вершину, которой ещё не было
- продолжаем пока не пройдём все вершины

```c++
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

typedef pair<int, int> pii;

void prim(int n, const vector<vector<pii>>& adj) {
    vector<bool> visited(n + 1, false);
    priority_queue<pii, vector<pii>, greater<pii>> pq;
    int mst_weight = 0;

    pq.push({0, 1}); // Начинаем с 1-й вершины

    while (!pq.empty()) {
        int w = pq.top().first;
        int u = pq.top().second;
        pq.pop();

        if (visited[u]) continue;

        visited[u] = true;
        mst_weight += w;

        for (auto& edge : adj[u]) {
            if (!visited[edge.first]) {
                pq.push({edge.second, edge.first});
            }
        }
    }
    cout << "Вес MST: " << mst_weight << endl;
}
```
