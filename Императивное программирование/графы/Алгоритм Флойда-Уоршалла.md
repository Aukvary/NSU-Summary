- находит кратчайшие пути между всеми парами вершин
- Для каждой пары вершин $(i, j)$ проверяется, не будет ли путь через промежуточную вершину $k$ короче, чем текущий известный путь
```c++
#include <iostream>
#include <vector>

using namespace std;

const int INF = 1e9;

void floydWarshall(int n, vector<vector<int>>& dist) {
    for (int k = 1; k <= n; k++) {
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                if (dist[i][k] < INF && dist[k][j] < INF) {
                    dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);
                }
            }
        }
    }
}
```
