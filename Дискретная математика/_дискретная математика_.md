## темы
```dataviewjs
const currentFile = dv.current().file;
const currentFolder = currentFile.path.split("/").slice(0, -1).join("/");
const basePath = currentFolder || "";

const folders = new Set();
for (let file of dv.pages().file) {
    if (file.path.startsWith(basePath) && file.path !== basePath) {
        const pathParts = file.path.substring(basePath.length + 1).split("/");
        if (pathParts[0] && !pathParts[0].includes(".")) {
            folders.add(pathParts[0]);
        }
    }
}

const sortedFolders = Array.from(folders).sort();

for (let folder of sortedFolders) {
    const folderPath = basePath ? `${basePath}/${folder}` : folder;
    
    const definitions = dv
	    .pages(`"${folderPath}" and #понятие`)
	    .sort(p => p.file.name, "asc");
	    
    const theoremsAndLemmas = dv
	    .pages(`"${folderPath}" and (#теорема or #лемма)`)
	    .sort(p => p.file.name, "asc");
        
    if (definitions.length + theoremsAndLemmas.length === 0) continue;


	dv.header(3, folder);
    
    if (definitions.length > 0) {
        dv.paragraph("- **определения:**");
        dv.list(definitions.file.link);
    }
    
    if (theoremsAndLemmas.length > 0) {
        dv.paragraph("- **теоремы и леммы:**");
        dv.list(theoremsAndLemmas.file.link)
    }
}
```

## условные обозначения
- $\in$ - принадлежит
- $:=$ - равно по определению
- $\forall$ - для всех
- $\exists$ - существует хотя бы одно
- $\subseteq$ - является подмножеством 
- <=>/<->- тогда и только тогда/равносильно
- =>/->- отсюда следует
- & - логическое и
- v - логическое или
- $\neg$ - логическое отрицание
- $\varnothing$ - пустое множество
- $\cap$ - пересечение
- $\cup$ - объединение
- $\triangle$ - симметричная разность 
- $\bar{x}$ - кортеж/набор элементов 
- $f: A \to B$ - отображение 
- $B^A$ - множество всех отображений
- $\overset{1:1}{\longrightarrow}$ - [[инъективное отображение]]
- $\twoheadrightarrow$ - [[сюръективное отображение]]
- $\rightarrowtail$ - [[всюду определённое отображение]]
- $\overset{1-1}{\twoheadrightarrow}$ - [[биектиное отображение]]
- $(f \circ g)(a)$ - [[композиция функций]]
- $[a]_{\sim}$ - [[класс эквивалентности]] 
- $A_{prop}$ - [[алфавит логики высказываний]]
- $(\varphi \leftrightarrow \psi)$ - $((\varphi \to \psi) \land (\psi \to \varphi))$
